using System;

class Program
{
    static void Main()
    {
        int[] ocene = new int[5];

        for (int i = 0; i < 5; i++)
        {
            Console.WriteLine("Unesite ocenu za kontrolni broj {0}:", i + 1);
            ocene[i] = Convert.ToInt32(Console.ReadLine());
        }

        Array.Sort(ocene);

        double prosecnaOcena = (ocene[1] + ocene[2] + ocene[3] + ocene[4]) / 4.0;
        Console.WriteLine("Prosečna ocena bez najlošije ocene: {0:F2}", prosecnaOcena);
    }
}
