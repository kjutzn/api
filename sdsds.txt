using System;

class Program
{
    static void Main()
    {
        int[] ocene = new int[5];
        double prosecnaoc;

        for (int i = 0; i < 5; i++)
        {
            ocene[i] = int.Parse(Console.ReadLine());
        }

        Array.Sort(ocene);

        prosecnaoc = (ocene[1] + ocene[2] + ocene[3] + ocene[4]) / 4.0;
        Console.WriteLine(prosecnaoc);
    }
}
